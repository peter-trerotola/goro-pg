go-postgres-mcp

read-only postgresql mcp server with schema intelligence

install
curl -sfL https://peter-trerotola.github.io/go-postgres-mcp/install.sh | sh

or

go install github.com/peter-trerotola/go-postgres-mcp/cmd@latest

or

docker pull ghcr.io/peter-trerotola/go-postgres-mcp:latest

linux/darwin · amd64/arm64

config
# interactive setup — discovers databases on your host, generates config.yaml
curl -sfL https://peter-trerotola.github.io/go-postgres-mcp/setup.sh | sh
# config.yaml — full discovery, all schemas and tables
databases:
  - name: "mydb"
    host: "localhost"
    port: 5432
    database: "mydb"
    user: "readonly_user"
    password_env: "DB_PASSWORD"

knowledgemap:
  path: "knowledgemap.db"
  auto_discover_on_startup: true
# config.yaml — narrow scope: specific schemas and tables only
databases:
  - name: "production"
    host: "db.example.com"
    port: 5432
    database: "app_prod"
    user: "mcp_reader"
    password_env: "PROD_DB_PASSWORD"
    sslmode: "require"

    # only discover these schemas
    schemas:
      - "public"
      - "billing"

    # only discover these tables
    tables:
      include:
        - "public.users"
        - "public.orders"
        - "billing.invoices"
        - "billing.line_items"

knowledgemap:
  path: "/data/knowledgemap.db"
  auto_discover_on_startup: true
